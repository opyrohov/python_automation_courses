## Компоненти

```python
# components/header.py
class Header:
    def __init__(self, page):
        self.page = page
        self.logo = page.get_by_role("link", name="Logo")
        self.search = page.get_by_placeholder("Search")
        self.cart = page.get_by_role("link", name="Cart")

    def search_for(self, query):
        self.search.fill(query)
        self.search.press("Enter")

# pages/home_page.py
class HomePage:
    def __init__(self, page):
        self.page = page
        self.header = Header(page)

    def search(self, query):
        self.header.search_for(query)
```

## Page Factory

```python
# pages/__init__.py
class Pages:
    def __init__(self, page):
        self.page = page
        self._login = None
        self._dashboard = None

    @property
    def login(self):
        if not self._login:
            self._login = LoginPage(self.page)
        return self._login

    @property
    def dashboard(self):
        if not self._dashboard:
            self._dashboard = DashboardPage(self.page)
        return self._dashboard

# Використання
def test_example(page):
    pages = Pages(page)
    pages.login.goto()
    pages.login.login("user", "pass")
    expect(pages.dashboard.welcome_message).to_be_visible()
```

## Base Page з helpers

```python
class BasePage:
    def __init__(self, page):
        self.page = page

    def wait_for_api(self, url_pattern):
        """Очікування API відповіді."""
        with self.page.expect_response(url_pattern) as response:
            yield
        return response.value

    def scroll_to(self, locator):
        """Прокрутка до елемента."""
        locator.scroll_into_view_if_needed()

    def take_screenshot(self, name):
        """Скріншот з ім'ям."""
        self.page.screenshot(path=f"screenshots/{name}.png")
```

## Data-driven POM

```python
from dataclasses import dataclass

@dataclass
class UserData:
    email: str
    password: str
    name: str = ""

class RegistrationPage(BasePage):
    def register(self, user: UserData):
        self.page.get_by_label("Email").fill(user.email)
        self.page.get_by_label("Password").fill(user.password)
        if user.name:
            self.page.get_by_label("Name").fill(user.name)
        self.page.get_by_role("button", name="Register").click()

# Тест
def test_registration(page):
    user = UserData(
        email="test@example.com",
        password="secret123",
        name="Test User"
    )
    RegistrationPage(page).register(user)
```
