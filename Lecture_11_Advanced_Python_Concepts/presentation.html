Part 1

List & Dictionary Comprehensions

Write elegant, efficient Python code

What Are Comprehensions?

Concise way to create lists, dictionaries, and sets from existing iterables

# Traditional loop squares = [] for x in range(5): squares.append(x ** 2) # List comprehension (one line!) squares = [x ** 2 for x in range(5)]
Benefits: More readable, faster execution, more Pythonic!

List Comprehension Syntax

# Basic syntax [expression for item in iterable] # With condition (filter) [expression for item in iterable if condition] # With if-else (transform) [expr_if_true if condition else expr_if_false for item in iterable]
# Examples evens = [x for x in range(10) if x % 2 == 0] # [0, 2, 4, 6, 8] labels = ["even" if x % 2 == 0 else "odd" for x in range(5)] # ['even', 'odd', 'even', 'odd', 'even']

Dictionary Comprehensions

# Basic syntax {key_expr: value_expr for item in iterable} # Example: Create squares dictionary squares = {x: x ** 2 for x in range(1, 6)} # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25} # Example: Zip two lists names = ["Alice", "Bob", "Charlie"] ages = [25, 30, 35] people = {name: age for name, age in zip(names, ages)} # {'Alice': 25, 'Bob': 30, 'Charlie': 35}
Use Case: Transform data, filter dictionaries, create lookups

Comprehensions in Test Automation

# Filter test data all_users = get_users() active_admins = [u for u in all_users if u["active"] and u["role"] == "admin"] # Generate test selectors button_ids = ["submit", "cancel", "delete"] selectors = [f"button#{btn}" for btn in button_ids] # Extract failed test names results = get_test_results() failed = [t["name"] for t in results if t["status"] == "failed"] # Generate test combinations browsers = ["chrome", "firefox"] envs = ["dev", "staging"] combos = [{"browser": b, "env": e} for b in browsers for e in envs]

Part 2

*args and **kwargs

Flexible function signatures

*args - Variable Positional Arguments

# Accept any number of positional arguments def sum_numbers(*args): return sum(args) print(sum_numbers(1, 2, 3)) # 6 print(sum_numbers(10, 20, 30, 40)) # 100 # *args is a tuple def print_args(*args): print(type(args)) # <class 'tuple'> for arg in args: print(arg)
Use Case: Functions that need to work with varying numbers of arguments

**kwargs - Variable Keyword Arguments

# Accept any number of keyword arguments def print_info(**kwargs): for key, value in kwargs.items(): print(f"{key}: {value}") print_info(name="Alice", age=25, city="NYC") # name: Alice # age: 25 # city: NYC # **kwargs is a dictionary def show_kwargs(**kwargs): print(type(kwargs)) # <class 'dict'> print(kwargs)

Combining Parameter Types

# Order matters! def flexible_function(required, *args, default="value", **kwargs): print(f"Required: {required}") print(f"Args: {args}") print(f"Default: {default}") print(f"Kwargs: {kwargs}") flexible_function( "must_have", "extra1", "extra2", default="custom", option1="val1", option2="val2" )
‚ö†Ô∏è Parameter Order:
1. Regular parameters
2. *args
3. Default parameters
4. **kwargs

Part 3

Decorators Basics

Extend function behavior elegantly

What Are Decorators?

Functions that modify or enhance other functions without changing their code

# Simple decorator def my_decorator(func): def wrapper(): print("Before function") func() print("After function") return wrapper # Apply decorator with @ syntax @my_decorator def say_hello(): print("Hello!") say_hello() # Output: # Before function # Hello! # After function

Decorators with Function Arguments

import functools def log_calls(func): @functools.wraps(func) # Preserve function metadata def wrapper(*args, **kwargs): print(f"Calling {func.__name__} with {args}, {kwargs}") result = func(*args, **kwargs) print(f"Result: {result}") return result return wrapper @log_calls def add(a, b): return a + b add(5, 3) # Calling add with (5, 3), {} # Result: 8
‚ö†Ô∏è Always use @functools.wraps to preserve function name and docstring!

Practical Decorator Examples

# Timing decorator def timer(func): @functools.wraps(func) def wrapper(*args, **kwargs): start = time.time() result = func(*args, **kwargs) duration = time.time() - start print(f"{func.__name__} took {duration:.2f}s") return result return wrapper # Retry decorator def retry(max_attempts=3): def decorator(func): @functools.wraps(func) def wrapper(*args, **kwargs): for attempt in range(max_attempts): try: return func(*args, **kwargs) except Exception as e: if attempt == max_attempts - 1: raise return wrapper return decorator

Decorators in Test Automation

# Screenshot on failure def screenshot_on_failure(func): @functools.wraps(func) def wrapper(*args, **kwargs): try: return func(*args, **kwargs) except Exception: page.screenshot(path=f"{func.__name__}_failed.png") raise return wrapper # Retry flaky tests @retry(max_attempts=3) @screenshot_on_failure def test_flaky_feature(): # Test code here pass
Use Cases: Retry logic, logging, timing, screenshots, cleanup, authentication

Working with Datetime

Handle dates and times like a pro

Datetime Module Basics

from datetime import datetime, date, timedelta # Current date and time now = datetime.now() today = date.today() # Create specific dates birthday = date(1990, 5, 15) meeting = datetime(2024, 12, 25, 14, 30) # Date components print(now.year, now.month, now.day) print(now.hour, now.minute, now.second)
# Date arithmetic tomorrow = today + timedelta(days=1) next_week = today + timedelta(weeks=1) future = now + timedelta(days=7, hours=3, minutes=30)

Formatting and Parsing Dates

# Format datetime to string (strftime) now = datetime.now() print(now.strftime("%Y-%m-%d")) # 2024-12-25 print(now.strftime("%m/%d/%Y")) # 12/25/2024 print(now.strftime("%B %d, %Y")) # December 25, 2024 print(now.strftime("%I:%M %p")) # 02:30 PM # Parse string to datetime (strptime) date_str = "2024-12-25" parsed = datetime.strptime(date_str, "%Y-%m-%d") date_str2 = "December 25, 2024" parsed2 = datetime.strptime(date_str2, "%B %d, %Y")
Common Format Codes:
%Y=year, %m=month, %d=day, %H=hour(24), %I=hour(12), %M=minute, %S=second, %p=AM/PM

Datetime in Test Automation

Split and Join

# SPLIT - Break string into list text = "Hello World Python" words = text.split() # Split on whitespace # ['Hello', 'World', 'Python'] csv = "Alice,25,NYC" fields = csv.split(",") # Split on comma # ['Alice', '25', 'NYC'] # JOIN - Combine list into string words = ["Hello", "World"] joined = " ".join(words) # Join with space # 'Hello World' path_parts = ["home", "user", "docs"] path = "/".join(path_parts) # Join with slash # 'home/user/docs'

Strip and Replace

Case Conversion Methods

text = "Hello World Python" text.upper() # 'HELLO WORLD PYTHON' text.lower() # 'hello world python' text.title() # 'Hello World Python' text.capitalize() # 'Hello world python' # Case-insensitive comparison name1 = "Alice" name2 = "ALICE" if name1.lower() == name2.lower(): print("Names match!")

Search and Check Methods

Example: Flexible URL Builder

def build_api_url(base, *paths, **params): """ Build API URL with paths and query parameters Example: build_api_url('api.test.com', 'users', 123, page=1, limit=10) Returns: 'api.test.com/users/123?page=1&limit=10' """ # Join base with paths url = base.rstrip("/") if paths: path_str = "/".join(str(p).strip("/") for p in paths) url += f"/{path_str}" # Add query parameters if params: query = "&".join(f"{k}={v}" for k, v in params.items()) url += f"?{query}" return url # Usage url = build_api_url( "https://api.example.com", "v1", "users", 123, page=1, limit=10, sort="name" )

def generate_report(results, **metadata): """Generate formatted test report""" lines = ["=" * 60, "TEST REPORT", "=" * 60] # Add timestamp timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S") lines.append(f"Generated: {timestamp}") lines.append("") # Add metadata if metadata: lines.append("Environment:") for key, value in metadata.items(): lines.append(f" {key.title()}: {value}") lines.append("") # Calculate statistics using comprehensions total = len(results) passed = sum(1 for r in results if r["status"] == "passed") duration = sum(r["duration"] for r in results) lines.append(f"Total: {total}, Passed: {passed}, Duration: {duration}s") return "\n".join(lines)

Best Practices

  • *args/**kwargs: Document expected parameters in docstrings
  • Decorators: Always use @functools.wraps to preserve metadata
  • Datetime: Use ISO format for consistency, be timezone-aware
  • Strings: Remember most methods return new strings (immutable)
  • Testing: Combine concepts for powerful automation helpers
