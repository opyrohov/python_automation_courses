Reading Files - Basics

# Open and read entire file with open('file.txt', 'r') as f: content = f.read() print(content) # Read line by line with open('file.txt', 'r') as f: for line in f: print(line.strip()) # Read all lines into a list with open('file.txt', 'r') as f: lines = f.readlines()
Always use with statement - it auto-closes the file!

Writing Files

# Write (overwrite) - mode 'w' with open('output.txt', 'w') as f: f.write("Hello, World!\n") f.write("Line 2\n") # Append - mode 'a' with open('output.txt', 'a') as f: f.write("Appended line\n") # Write multiple lines lines = ["Line 1\n", "Line 2\n"] with open('output.txt', 'w') as f: f.writelines(lines)
'w' = overwrite   |   'a' = append

File Modes

# Common file modes 'r' # Read (default) 'w' # Write (overwrite) 'a' # Append 'x' # Exclusive creation (fails if exists) 'r+' # Read and write 'b' # Binary mode (rb, wb)

Text Mode

  • Default mode
  • Handles encoding
  • For .txt, .json, .csv

Binary Mode

  • Add 'b' to mode
  • For images, PDFs
  • No encoding

Why Use Context Managers?

# ‚ùå Without context manager (bad!) f = open('file.txt', 'r') content = f.read() f.close() # Easy to forget! # ‚úÖ With context manager (good!) with open('file.txt', 'r') as f: content = f.read() # File auto-closes here!
Benefits:
  • ‚úÖ Automatic cleanup
  • ‚úÖ Handles exceptions
  • ‚úÖ Cleaner code
  • ‚úÖ Prevents resource leaks

What is JSON?

JSON (JavaScript Object Notation) is a lightweight data format perfect for configuration and test data.
{ "name": "Alice", "age": 25, "active": true, "roles": ["user", "tester"], "config": { "theme": "dark", "notifications": true } }
Python ‚Üî JSON:
dict ‚Üí object | list ‚Üí array | True/False ‚Üí true/false

Reading JSON Files

import json # Read JSON from file with open('data.json', 'r') as f: data = json.load(f) print(data['name']) # Alice print(data['age']) # 25 # Parse JSON string json_str = '{"name": "Bob", "age": 30}' data = json.loads(json_str) print(data['name']) # Bob
load = from file   |   loads = from string

Writing JSON Files

import json # Create data data = { "username": "testuser", "email": "test@example.com", "age": 25, "active": True } # Write to file with open('user.json', 'w') as f: json.dump(data, f, indent=2) # Convert to string json_str = json.dumps(data, indent=2) print(json_str)
dump = to file   |   dumps = to string
indent=2 makes it pretty!

JSON for Test Data

# users.json - Test users file { "users": [ { "username": "testuser1", "password": "Test123!", "role": "user" }, { "username": "admin", "password": "Admin456!", "role": "admin" } ] } # Load and use in tests with open('users.json') as f: users = json.load(f)['users'] for user in users: test_login(user['username'], user['password'])

JSON for Configuration

# config.json { "environment": "staging", "base_url": "https://staging.example.com", "browser": { "type": "chromium", "headless": false }, "timeouts": { "default": 30000, "navigation": 60000 } } # Load configuration with open('config.json') as f: config = json.load(f) page.goto(config['base_url'])

What is CSV?

CSV (Comma-Separated Values) is perfect for tabular test data.
# test_cases.csv test_id,test_name,username,password,expected 1,Valid Login,user@test.com,Test123!,success 2,Invalid Password,user@test.com,wrong,failure 3,Empty Username,,Test123!,failure 4,Empty Password,user@test.com,,failure
Use CSV for:
  • Data-driven testing
  • Test cases with parameters
  • Test reports

Reading CSV Files

import csv # Method 1: Basic reader with open('data.csv', 'r') as f: reader = csv.reader(f) for row in reader: print(row) # Method 2: DictReader (better!) with open('data.csv', 'r') as f: reader = csv.DictReader(f) for row in reader: print(row['username']) print(row['password'])
DictReader gives you dictionaries - much easier to use!

Writing CSV Files

import csv # Write with DictWriter data = [ {'name': 'Alice', 'email': 'alice@test.com'}, {'name': 'Bob', 'email': 'bob@test.com'} ] with open('users.csv', 'w', newline='') as f: fieldnames = ['name', 'email'] writer = csv.DictWriter(f, fieldnames=fieldnames) writer.writeheader() writer.writerows(data)
Note: Use newline='' to prevent blank lines on Windows

Data-Driven Testing with CSV

import csv # test_cases.csv has test scenarios with open('test_cases.csv') as f: reader = csv.DictReader(f) for test_case in reader: # Run test with this data page.goto("https://example.com/login") page.fill("#username", test_case['username']) page.fill("#password", test_case['password']) page.click("#login-button") # Verify result matches expected if test_case['expected'] == 'success': assert page.url.endswith('/dashboard')

Working with Paths - pathlib

from pathlib import Path # Current file location current_file = Path(__file__) print(current_file.name) # File name print(current_file.parent) # Parent directory # Build paths (works on any OS!) project_root = Path("/project") data_file = project_root / "test_data" / "users.json" # Check properties if data_file.exists(): print("File exists!")
pathlib is cross-platform and object-oriented!

Writing Log Files

from datetime import datetime def log_message(message, level="INFO"): """Write log message with timestamp.""" timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S") log_entry = f"[{timestamp}] [{level}] {message}\n" with open('test.log', 'a') as f: f.write(log_entry) # Usage log_message("Test started") log_message("Login successful", "SUCCESS") log_message("Test failed", "ERROR")

Creating Directories

File Handling Best Practices

‚úÖ DO

  • Use with statement
  • Use pathlib for paths
  • Check file existence
  • Handle exceptions
  • Use JSON for config
  • Use CSV for test data

‚ùå DON'T

  • Forget to close files
  • Hardcode paths
  • Ignore encoding
  • Store secrets in files
  • Commit output files

Error Handling

from pathlib import Path def safe_read_json(file_path): """Safely read JSON file with error handling.""" file_path = Path(file_path) if not file_path.exists(): print(f"File not found: {file_path}") return None try: with open(file_path) as f: return json.load(f) except json.JSONDecodeError: print("Invalid JSON format") return None
